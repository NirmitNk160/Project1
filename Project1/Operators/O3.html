    function compareValues() {
      let xRaw = document.getElementById('xInput').value;
      let yRaw = document.getElementById('yInput').value;
      const resultElem = document.getElementById('result');
      // Try to convert x and y to number or boolean where appropriate
      function convert(val) {
        if(val.trim().toLowerCase() === "true") return true;
        else if(val.trim().toLowerCase() === "false") return false;
        else if(!isNaN(Number(val)) && val.trim() !== "") return Number(val);
        else return val;
      }
      let x = convert(xRaw);
      let y = convert(yRaw);
      let output = "";
      output += `x == y : ${x == y}\n`;          // loose equality
      output += `x === y : ${x === y}\n`;        // strict equality
      output += `x != y : ${x != y}\n`;          // loose inequality
      output += `x !== y : ${x !== y}\n`;        // strict inequality
      output += `x > y : ${x > y}\n`;
      output += `x < y : ${x < y}\n`;
      output += `x >= y : ${x >= y}\n`;
      output += `x <= y : ${x <= y}\n`;
      resultElem.textContent = output;
    }
  </script>

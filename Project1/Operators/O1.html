  <button onclick="calculate()">Calculate</button>

  <pre id="result"></pre>

  <script>
    function calculate() {
      const a = Number(document.getElementById('inputA').value);
      const b = Number(document.getElementById('inputB').value);
      const resultEl = document.getElementById('result');

      if (isNaN(a) || isNaN(b)) {
        resultEl.textContent = "Please enter valid numbers for both a and b.";
        return;
      }

      let output = "";
      output += `a + b = ${a + b}\n`;
      output += `a - b = ${a - b}\n`;
      output += `a * b = ${a * b}\n`;
      output += `a / b = ${b !== 0 ? (a / b) : 'Division by zero error'}\n`;
      output += `a % b = ${b !== 0 ? (a % b) : 'Modulo by zero error'}\n`;
      output += `a ** 2 = ${a ** 2}\n`;

      resultEl.textContent = output;
    }
  </script>

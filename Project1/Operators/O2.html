    <input class="input" type="number" id="inputA" placeholder="Enter initial value of a"><br><br>
    <label>Enter value to add, multiply, subtract, and divide:</label>
    <input class="input" type="number" id="inputB" placeholder="Enter value for operations"><br><br>
    <button onclick="calculate()">Calculate</button>
    <pre id="result"></pre>

  <script>
    function calculate() {
      let a = Number(document.getElementById('inputA').value);
      const b = Number(document.getElementById('inputB').value);
      const resultEl = document.getElementById('result');
      if (isNaN(a) || isNaN(b)) {
        resultEl.textContent = "Please enter valid numbers for both inputs.";
        return;
      }
      let output = `Initial a = ${a}\n`;
      a += b;
      output += `After a += b: ${a}\n`;
      a *= b;
      output += `After a *= b: ${a}\n`;
      a -= b;
      output += `After a -= b: ${a}\n`;
      if(b === 0) {
        output += `After a /= b: Division by zero error\n`;
      } else {
        a /= b;
        output += `After a /= b: ${a}\n`;
      }
      resultEl.textContent = output;
    }
  </script>

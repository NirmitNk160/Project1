  <script>
    function calculateLogical() {
      const aRaw = document.getElementById('aInput').value.trim().toLowerCase();
      const bRaw = document.getElementById('bInput').value.trim().toLowerCase();
      const resultEl = document.getElementById('result');
      const toBoolean = (val) => {
        if (val === 'true') return true;
        if (val === 'false') return false;
        return null; // invalid input
      };
      const a = toBoolean(aRaw);
      const b = toBoolean(bRaw);
      if (a === null || b === null) {
        resultEl.textContent = "Please enter 'true' or 'false' for both a and b.";
        return;
      }
      let output = "";
      output += `a && b : ${a && b}\n`;
      output += `a || b : ${a || b}\n`;
      output += `!a : ${!a}\n`;
      resultEl.textContent = output;
    }
  </script>

  <input type="number" class="input" id="ageInput" placeholder="Enter age" ><br><br><br>
  <button onclick="checkStatus()">Check Status</button>
  <p id="output"></p>

  <script>
    function checkStatus() {
      let age = Number(document.getElementById('ageInput').value);
      if (isNaN(age) || age < 0) {
        document.getElementById('output').innerText = "Please enter a valid age.";
        return;
      }
      let status = age >= 18 ? "Adult" : "Minor";
      document.getElementById('output').innerText = "Age: " + age + "\nStatus: " + status;
    }
  </script>

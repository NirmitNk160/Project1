<h2>Evaluate grade from score</h2>
<input class="input" type="number" id="scoreInput" placeholder="Enter score (0-100)"><br><br>
<button onclick="evaluateGrade()">Get Grade</button>
<p id="result"></p>

<script>
  function evaluateGrade() {
    let score = Number(document.getElementById('scoreInput').value);
    let grade;
    if(score >= 90) {
      grade = "A";
    } else if(score >= 75) {
      grade = "B";
    } else if(score >= 60) {
      grade = "C";
    } else {
      grade = "F";
    }
    document.getElementById('result').innerText = "Grade: " + grade;
  }
</script>

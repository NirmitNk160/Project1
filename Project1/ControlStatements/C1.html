<h2>Check positive, negative, or zero</h2>
<label>Enter a number: <input class="input" type="number" id="numInput" placeholder="Enter a number"></label><br><br>
<button onclick="checkNumber()">Check</button>
<p id="result"></p>

<script>
  function checkNumber() {
    let num = Number(document.getElementById('numInput').value);
    let output = "";
    if(num > 0) {
     output = num + " is positive";
    } else if(num < 0) {
     output = num + " is negative";
    } else {
     output = "The number is zero";
    }
    document.getElementById('result').innerText = output;
  }
</script>

<input class="input" type="number" id="ageInput" placeholder="Enter your age"><br><br>
<button onclick="checkEligibility()">Check</button>
<p id="result"></p>

<script>
  function checkEligibility() {
    let age = Number(document.getElementById('ageInput').value);
    let eligible = (age >= 18) ? "Eligible to vote" : "Not eligible to vote";
    document.getElementById('result').innerText = "Age: " + age + " - " + eligible;
  }
</script>

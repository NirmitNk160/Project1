<p id="result"></p>

<script>
  function checkTemperature() {
    let temp = Number(document.getElementById('tempInput').value);
    let message;
    if(temp > 30) {
      if(temp > 40) {
        message = "It's extremely hot";
      } else {
        message = "It's hot";
      }
    } else {
      message = "It's cool";
    }
    document.getElementById('result').innerText = message;
  }
</script>

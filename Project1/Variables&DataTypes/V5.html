  <h2>Using const and let variables in operations:</h2>
  <script>
    const pi = 3.14;
    let radius = 5;
    let area = pi * radius * radius;

    document.write("Radius: " + radius + "<br>");
    document.write("Area of circle: " + area + "<br>");

    radius = 7; // changing variable
    area = pi * radius * radius;
    document.write("New radius: " + radius + "<br>");
    document.write("Updated area: " + area + "<br>");
  </script>

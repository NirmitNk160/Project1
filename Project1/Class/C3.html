  <h2>Rectangle Area and Perimeter</h2>
  <input class="input" type="number" id="width" placeholder="Width"><br><br>
  <input class="input" type="number" id="height" placeholder="Height"><br><br>
  <button onclick="calculate()">Calculate</button>
  <p id="output"></p>

  <script>
    class Rectangle {
      constructor(width, height) {
        this.width = width;
        this.height = height;
      }
      area() {
        return this.width * this.height;
      }
      perimeter() {
        return 2 * (this.width + this.height);
      }
    }

    function calculate() {
      const w = Number(document.getElementById('width').value);
      const h = Number(document.getElementById('height').value);
      if(isNaN(w) || isNaN(h) || w <= 0 || h <= 0) {
        alert("Enter valid positive numbers");
        return;
      }
      const rect = new Rectangle(w, h);
      const output = `Area: ${rect.area()}\nPerimeter: ${rect.perimeter()}`;
      document.getElementById('output').innerText = output;
    }
  </script>

  <h2>Student Grade Calculator</h2>
  <input class="input" type="text" id="studentName" placeholder="Student Name"><br><br>
  <input class="input" type="number" id="marks" placeholder="Marks (0-100)"><br><br>
  <button onclick="showGrade()">Calculate Grade</button>
  <p id="output"></p>

  <script>
    class Student {
      constructor(name, marks) {
        this.name = name;
        this.marks = marks;
      }
      getGrade() {
        if(this.marks >= 90) return 'A';
        else if(this.marks >= 75) return 'B';
        else if(this.marks >= 60) return 'C';
        else if(this.marks >= 40) return 'D';
        else return 'F';
      }
    }

    function showGrade() {
      const name = document.getElementById('studentName').value;
      const marks = Number(document.getElementById('marks').value);
      if(name === "" || isNaN(marks) || marks < 0 || marks > 100) {
        alert("Enter valid name and marks");
        return;
      }
      const student = new Student(name, marks);
      const output = `Name: ${student.name}\nMarks: ${student.marks}\nGrade: ${student.getGrade()}`;
      document.getElementById('output').innerText = output;
    }
  </script>

  <h2>Calculator: Add/Subtract</h2>
  <input class="input" type="number" id="num1" placeholder="Number 1"><br><br>
  <input class="input" type="number" id="num2" placeholder="Number 2"><br><br>
  <button onclick="calculate()">Calculate</button>
  <p id="output"></p>

  <script>
    class Calculator {
      add(a, b) { return a + b; }
      subtract(a, b) { return a - b; }
    }

    function calculate() {
      const a = Number(document.getElementById('num1').value);
      const b = Number(document.getElementById('num2').value);
      if(isNaN(a) || isNaN(b)) {
        alert("Enter valid numbers");
        return;
      }
      const calc = new Calculator();
      const output = `Sum: ${calc.add(a,b)}\nDifference: ${calc.subtract(a,b)}`;
      document.getElementById('output').innerText = output;
    }
  </script>

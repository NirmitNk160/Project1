  <h2>Create Person and Show Info</h2>
  <input class="input" type="text" id="nameInput" placeholder="Name"><br><br>
  <input class="input" type="number" id="ageInput" placeholder="Age"><br><br>
  <button onclick="createPerson()">Create Person</button>
  <p id="output"></p>

  <script>
    class Person {
      constructor(name, age) {
        this.name = name;
        this.age = age;
      }
      info() {
        return `Name: ${this.name}, Age: ${this.age}`;
      }
    }

    function createPerson() {
      const name = document.getElementById('nameInput').value;
      const age = Number(document.getElementById('ageInput').value);
      if(name === "" || isNaN(age)) {
        alert("Please enter valid name and age");
        return;
      }
      const person = new Person(name, age);
      document.getElementById('output').innerText = person.info();
    }
  </script>

Program to calculate sum of numbers from 1 to n using while loop

<body>
  <h2>Calculate sum from 1 to n</h2>
  <input class="input" type="number" id="sumInput" placeholder="Enter n"><br><br>
  <button onclick="showSum()">Show Sum</button>
  <p id="output"></p>
  <script>
    function showSum() {
      let n = Number(document.getElementById('sumInput').value);
      let sum = 0, i = 1;
      while(i <= n) {
        sum += i;
        i++;
      }
      document.getElementById('output').textContent = "Sum: " + sum;
    }
  </script>

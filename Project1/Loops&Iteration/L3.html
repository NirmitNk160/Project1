  <h2>Show each character of string</h2>
  <input class="input" type="text" id="strInput" placeholder="Enter a string"><br><br>
  <button onclick="printChars()">Show Characters</button>
  <pre id="output"></pre>
  <script>
    function printChars() {
      let str = document.getElementById('strInput').value;
      let output = "", i = 0;
      do {
        if(str.length === 0) break;
        output += str[i] + "\n";
        i++;
      } while(i < str.length);
      document.getElementById('output').textContent = output.trim();
    }
  </script>

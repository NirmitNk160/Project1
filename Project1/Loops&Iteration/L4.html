<body>
  <h2>Show Array Items (for...of)</h2>
  <input class="input" type="text" id="arrInput" placeholder="Item1,Item2,Item3"><br><br>
  <button onclick="showItems()">Show</button>
  <p id="result"></p>
  <script>
    function showItems() {
      let arr = document.getElementById('arrInput').value.split(',');
      let out = "";
      for(let item of arr) {
        out += item.trim() + " ";
      }
      document.getElementById('result').innerText = out.trim();
    }
  </script>

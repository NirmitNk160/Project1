  <input class="input" type="text" id="nameInput" placeholder="Name"><br><br>
  <input class="input" type="number" id="ageInput" placeholder="Age"><br><br>
  <input class="input" type="text" id="cityInput" placeholder="City"><br><br>
  <button onclick="showObject()">Show Object</button>
  <pre id="output"></pre>
  <script>
    function showObject() {
      let obj = {
        name: document.getElementById('nameInput').value,
        age: document.getElementById('ageInput').value,
        city: document.getElementById('cityInput').value
      };
      let output = "";
      for(let key in obj) {
        output += key + ": " + obj[key] + "\n";
      }
      document.getElementById('output').textContent = output.trim();
    }
  </script>

  <h2>Numbers from 1 to n</h2>
  <input class="input" type="number" id="nInput" placeholder="Enter n"><br><br>
  <button onclick="printNumbers()">Go</button>
  <p id="result"></p>
  <script>
    function printNumbers() {
      let n = Number(document.getElementById('nInput').value);
      let out = "";
      for(let i=1; i<=n; i++) {
        out += i + " ";
      }
      document.getElementById('result').innerText = out;
    }
  </script>

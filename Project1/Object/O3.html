  <p>Object: <span id="objDisplay"></span></p>
  <input class="input" type="text" id="propInput" placeholder="Property name to delete"><br><br>
  <button onclick="deleteProperty()">Delete Property</button>

  <script>
    let person = {name: "Nirmit", age: 17, city: "Kalburgi"};

    function updateDisplay() {
      document.getElementById('objDisplay').innerText = JSON.stringify(person);
    }

    function deleteProperty() {
      const prop = document.getElementById('propInput').value.trim();
      if(prop in person) {
        delete person[prop];
        updateDisplay();
      } else {
        alert("Property does not exist");
      }
    }

    updateDisplay();
  </script>

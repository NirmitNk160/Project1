  <input class="input" type="number" id="ageInput" placeholder="Age"><br><br>
  <input class="input" type="text" id="cityInput" placeholder="City"><br><br>
  <button onclick="createObject()">Create Object</button>
  <p id="output"></p>

  <script>
    function createObject() {
      let person = {
        name: document.getElementById('nameInput').value,
        age: Number(document.getElementById('ageInput').value),
        city: document.getElementById('cityInput').value
      };
      document.getElementById('output').innerText = 
        `Name: ${person.name}\nAge: ${person.age}\nCity: ${person.city}`;
    }

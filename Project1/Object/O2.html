  <h2>Update Person's City</h2>
  <p>Current city: <span id="cityDisplay">Kalburgi</span></p>
  <input class="input" type="text" id="newCityInput" placeholder="Enter new city"><br><br>
  <button onclick="updateCity()">Update City</button>

  <script>
    let person = {name: "John", age: 30, city: "New York"};

    function updateCity() {
      const newCity = document.getElementById('newCityInput').value;
      if(newCity.trim() === "") return;
      person.city = newCity;
      document.getElementById('cityDisplay').innerText = person.city;
    }
  </script>

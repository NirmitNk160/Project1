  <h2>Check if Object has Property</h2>
  <input class="input" type="text" id="checkPropInput" placeholder="Property name to check"><br><br>
  <button onclick="checkProperty()">Check</button>
  <p id="result"></p>

  <script>
    let person = {name: "Emma", age: 28, city: "Chicago"};

    function checkProperty() {
      const prop = document.getElementById('checkPropInput').value.trim();
      const result = document.getElementById('result');
      if(prop === "") {
        result.innerText = "Please enter a property name.";
        return;
      }
      if(person.hasOwnProperty(prop)) {
        result.innerText = `Property "${prop}" exists in the object.`;
      } else {
        result.innerText = `Property "${prop}" does NOT exist in the object.`;
      }
    }
  </script>

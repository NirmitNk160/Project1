  <input class="input" type="text" id="strInput" placeholder="Enter text"><br><br>
  <button onclick="showReverse()">Reverse</button>
  <p id="output"></p>
  <script>
    function reverseString(s) {
      return s.split("").reverse().join("");
    }
    function showReverse() {
      const s = document.getElementById('strInput').value;
      document.getElementById('output').innerText = "Reversed: " + reverseString(s);
    }
  </script>

  <input class="input" type="number" id="factorInput" placeholder="Enter a number"><br><br>
  <button onclick="showFact()">Calculate</button>
  <p id="output"></p>
  <script>
    function factorial(n) {
      let f = 1;
      for(let i = 1; i <= n; i++) {
        f *= i;
      }
      return f;
    }
    function showFact() {
      const n = Number(document.getElementById('factorInput').value);
      document.getElementById('output').innerText = "Factorial: " + factorial(n);
    }
  </script>

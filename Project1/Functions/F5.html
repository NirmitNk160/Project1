  <input class="input" type="number" id="numC" placeholder="Third number"><br><br>
  <button onclick="showMax()">Find Max</button>
  <p id="output"></p>
  <script>
    function maxOfThree(a, b, c) {
      return Math.max(a, b, c);
    }
    function showMax() {
      const a = Number(document.getElementById('numA').value);
      const b = Number(document.getElementById('numB').value);
      const c = Number(document.getElementById('numC').value);
      document.getElementById('output').innerText = "Maximum: " + maxOfThree(a, b, c);
    }

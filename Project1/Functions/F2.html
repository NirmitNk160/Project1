  <input class="input" type="number" id="val" placeholder="Enter a number"><br><br>
  <button onclick="showEvenOdd()">Check</button>
  <p id="output"></p>
  <script>
    function isEven(n) {
      return n % 2 === 0 ? "Even" : "Odd";
    }
    function showEvenOdd() {
      const n = Number(document.getElementById('val').value);
      document.getElementById('output').innerText = n + " is " + isEven(n);
    }
  </script>

  <input class="input" type="number" id="num2" placeholder="Second number"><br><br>
  <button onclick="showSum()">Add</button>
  <p id="output"></p>
  <script>
    function add(a, b) {
      return a + b;
    }
    function showSum() {
      const a = Number(document.getElementById('num1').value);
      const b = Number(document.getElementById('num2').value);
      document.getElementById('output').innerText = "Sum: " + add(a, b);
    }
  </script>

  <script>
    let arr = ['orange', 'apple', 'banana', 'grape'];
    function displayList(listID, items) {
      const list = document.getElementById(listID);
      list.innerHTML = '';
      items.forEach(item => {
        const li = document.createElement('li');
        li.textContent = item;
        list.appendChild(li);
      });
    }
    function sortArray() {
      displayList('beforeSort', arr);
      let sorted = [...arr].sort();
      displayList('afterSort', sorted);
    }
    sortArray();
  </script>

  <input class="input" type="number" id="indexInput" placeholder="Enter index to remove"><br><br>
  <button onclick="removeItem()">Remove</button>
  <p>Current array:</p>
  <ul id="arrayList"></ul>


  <script>
    let arr = ['apple', 'banana', 'cherry' , 'mango' , 'orange','dragon fruit' , 'kivi'];
    function displayArray() {
      const list = document.getElementById('arrayList');
      list.innerHTML = "";
      arr.forEach((el, idx) => {
        let li = document.createElement('li');
        li.textContent = `${idx}: ${el}`;
        list.appendChild(li);
      });
    }
    function removeItem() {
      let index = Number(document.getElementById('indexInput').value);
      if(index >= 0 && index < arr.length) {
        arr.splice(index,1);
        displayArray();
      } else {
        alert("Invalid index");
      }
    }
    displayArray();
  </script>

  <input class="input" type="text" id="itemInput" placeholder="Enter item"><br><br>
  <button onclick="addItem()">Add Item</button>
  <p>Array contents:</p>
  <ul id="arrayList"></ul>


  <script>
    let arr = [];
    function addItem() {
      const item = document.getElementById('itemInput').value.trim();
      if(item === "") return;
      arr.push(item);
      document.getElementById('itemInput').value = "";
      displayArray();
    }
    function displayArray() {
      const list = document.getElementById('arrayList');
      list.innerHTML = "";
      arr.forEach((el, idx) => {
        let li = document.createElement('li');
        li.textContent = `${idx}: ${el}`;
        list.appendChild(li);
      });
    }
  </script>

  <h2>Search for an Item</h2>
  <input class="input" type="text" id="searchInput" placeholder="Enter item to search"> <br><br>
  <button onclick="searchItem()">Search</button>
  <p id="result"></p>


  <script>
    let arr = ['dog', 'cat', 'rabbit', 'parrot'];
    function searchItem() {
      let search = document.getElementById('searchInput').value.trim();
      let index = arr.indexOf(search);
      let result = document.getElementById('result');
      if(index !== -1) {
        result.textContent = `"${search}" found at index ${index}`;
      } else {
        result.textContent = `"${search}" not found in the array`;
      }
    }
    
  </script>

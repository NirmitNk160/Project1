  <script>
    let arr = ['one', 'two', 'three', 'four'];
    function displayList(listID, items) {
      const list = document.getElementById(listID);
      list.innerHTML = '';
      items.forEach(item => {
        const li = document.createElement('li');
        li.textContent = item;
        list.appendChild(li);
      });
    }
    function reverseArray() {
      displayList('originalList', arr);
      displayList('reversedList', [...arr].reverse());
    }
    reverseArray();
  </script>
